#this is CMakeLists for pnp_bridge

if(POLICY CMP0042)
    cmake_policy(SET CMP0042 NEW)
endif()

    add_subdirectory(impinj_reader)
endif()

    add_subdirectory(bluetooth_sensor)
    add_subdirectory(camera)
    add_subdirectory(core_device_health)
ENDIF(WIN32)
